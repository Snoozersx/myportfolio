#Finding Austin in dataset
df_austin <- df %>% 
  filter(city == "Austin")

# Setting the linear regression model
lm_model <- lm(log(sales) ~ listings + year + median, data = df_austin)

# Extracting the p-values and significant predictors
p_values <- summary(lm_model)$coefficients[, "Pr(>|t|)"]
significant_predictors <- p_values < 0.05
significant_predictors
## (Intercept)    listings        year      median 
##        TRUE        TRUE        TRUE        TRUE
#We can now see that all the predictors are significant!

# Checking the coef of the year to find if there is increase in sales over time
coefficient_year <- coef(lm_model)["year"]
increase_sales_over_time <- coefficient_year > 0

significant_predictors
## (Intercept)    listings        year      median 
##        TRUE        TRUE        TRUE        TRUE
increase_sales_over_time
##  year 
## FALSE

Therefore we find that there is not an increase in the sales of Austin over time. Although we the significant predictors have a statistical significance, the predictors do not indicate the significance in sales over time.
